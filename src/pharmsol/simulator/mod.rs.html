19/// This closure represents the differential equation of the model.
20///
21/// # Parameters
22/// - `x`: The state vector at time t
23/// - `p`: The parameters of the model; Use the [fetch_params!] macro to extract the parameters
24/// - `t`: The time at which the differential equation is evaluated
25/// - `dx`: A mutable reference to the derivative of the state vector at time t
26/// - `rateiv`: A vector of infusion rates at time t
27/// - `cov`: A reference to the covariates at time t; Use the [fetch_cov!] macro to extract the covariates
28///
29/// # Example
30/// ```ignore
31/// use pharmsol::*;
32/// let diff_eq = |x, p, t, dx, rateiv, cov| {
33///  fetch_params!(p, ka, ke, v);
34///  fetch_cov!(cov, t, wt);
35///  dx[0] = -ka * x[0];
36///  dx[1] = ka * x[0] - ke * x[1];
37/// };
38/// ```
39pub type DiffEq = fn(&V, &V, T, &mut V, V, &Covariates);

54/// This closure represents the drift term of a stochastic differential equation model.
55///
56/// # Parameters
57/// - `x`: The state vector at time t
58/// - `p`: The parameters of the model; Use the [fetch_params!] macro to extract the parameters
59/// - `t`: The time at which the drift term is evaluated
60/// - `dx`: A mutable reference to the derivative of the state vector at time t
61/// - `rateiv`: A vector of infusion rates at time t
62/// - `cov`: A reference to the covariates at time t; Use the [fetch_cov!] macro to extract the covariates
63///
64/// # Example
65/// ```ignore
66/// use pharmsol::*;
67/// let drift = |x, p, t, dx, rateiv, cov| {
68///   fetch_params!(p, mka, mke, v);
69///   fetch_cov!(cov, t, wt);
70///   ka = dx[2];
71///   ke = dx[3];
72///   dx[0] = -ka * x[0];
73///   dx[1] = ka * x[0] - ke * x[1];
74///   dx[2] = -dx[2] + mka; // Mean reverting to mka
75///   dx[3] = -dx[3] + mke; // Mean reverting to mke
76/// };
77/// ```
78pub type Drift = DiffEq;

88/// This closure represents the initial state of the system.
89///
90/// # Parameters
91/// - `p`: The parameters of the model; Use the [fetch_params!] macro to extract the parameters
92/// - `t`: The time at which the initial state is evaluated; Hardcoded to 0.0
93/// - `cov`: A reference to the covariates at time t; Use the [fetch_cov!] macro to extract the covariates
94/// - `x`: A mutable reference to the state vector at time t
95///
96/// # Example
97/// ```ignore
98/// use pharmsol::*;
99/// let init = |p, t, cov, x| {
100///  fetch_params!(p, ka, ke, v);
101///  fetch_cov!(cov, t, wt);
102///  x[0] = 500.0;
103///  x[1] = 0.0;
104/// };
105/// ```
106pub type Init = fn(&V, T, &Covariates, &mut V);

147/// This closure represents the lag time of the model.
148///
149/// The lag term delays only the boluses going into a specific compartment.
150///
151/// # Parameters
152/// - `p`: The parameters of the model; Use the [fetch_params!] macro to extract the parameters
153///
154/// # Returns
155/// - A hashmap with the lag times for each compartment. If not present, lag is assumed to be 0.
156///
157/// There is a convenience macro [lag!] to create the hashmap
158///
159/// # Example
160/// ```ignore
161/// use pharmsol::*;
162/// let lag = |p| {
163///    fetch_params!(p, tlag);
164///    lag! {0=>tlag, 1=>0.3}
165/// };
166/// ```
167/// This will lag the bolus going into the first compartment by tlag and the bolus going into the
168/// second compartment by 0.3
169pub type Lag = fn(&V) -> HashMap<usize, T>;

171/// This closure represents the fraction absorbed (also called bioavailability or protein binding)
172/// of the model.
173///
174/// The fa term is used to adjust the amount of drug that is absorbed into the system.
175///
176/// # Parameters
177/// - `p`: The parameters of the model; Use the [fetch_params!] macro to extract the parameters
178///
179/// # Returns
180/// - A hashmap with the fraction absorbed for each compartment. If not present, it is assumed to be 1.
181///
182/// There is a convenience macro [fa!] to create the hashmap
183///
184/// # Example
185/// ```ignore
186/// use pharmsol::*;
187/// let fa = |p| {
188///   fetch_params!(p, fa);
189///   fa! {0=>fa, 1=>0.3}
190/// };
191/// ```
192/// This will adjust the amount of drug absorbed into the first compartment by fa and the amount of drug
193/// absorbed into the second compartment by 0.3
194pub type Fa = fn(&V) -> HashMap<usize, T>;
