10/// Implementation of the Euler-Maruyama method for solving stochastic differential equations.
11///
12/// This structure holds the SDE system parameters and state, providing a numerical method
13/// for approximating solutions to stochastic differential equations with adaptive step size
14/// control for improved accuracy.
15#[derive(Clone)]
16pub struct EM {
17    drift: Drift,
18    diffusion: Diffusion,
19    params: DVector<f64>,
20    state: DVector<f64>,
21    cov: Covariates,
22    infusions: Vec<Infusion>,
23    rtol: f64,
24    atol: f64,
25    max_step: f64,
26    min_step: f64,
27}
28
29impl EM {
30    /// Creates a new SDE solver using the Euler-Maruyama method.
31    ///
32    /// # Arguments
33    ///
34    /// * `drift` - Function defining the deterministic component of the SDE
35    /// * `diffusion` - Function defining the stochastic component of the SDE
36    /// * `params` - Vector of model parameters
37    /// * `initial_state` - Initial state vector of the system
38    /// * `cov` - Covariates that may influence the system dynamics
39    /// * `infusions` - Vector of infusion events to be applied during simulation
40    /// * `rtol` - Relative tolerance for adaptive step size control
41    /// * `atol` - Absolute tolerance for adaptive step size control
42    ///
43    /// # Returns
44    ///
45    /// A new instance of the Euler-Maruyama solver configured with the given parameters.
46    pub fn new(
47        drift: Drift,
48        diffusion: Diffusion,
49        params: DVector<f64>,
50        initial_state: DVector<f64>,
51        cov: Covariates,
52        infusions: Vec<Infusion>,
53        rtol: f64,
54        atol: f64,
55    ) -> Self {
56        Self {
57            drift,
58            diffusion,
59            params,
60            state: initial_state,
61            cov,
62            infusions,
63            rtol,
64            atol,
65            max_step: 0.1,  // Can be made configurable
66            min_step: 1e-6, // Can be made configurable
67        }
68    }
69
70    /// Calculates the error between two approximations for adaptive step size control.
71    ///
72    /// # Arguments
73    ///
74    /// * `y1` - First approximation of the solution
75    /// * `y2` - Second approximation of the solution (typically more accurate)
76    ///
77    /// # Returns
78    ///
79    /// The maximum normalized error between the two approximations.
80    fn calculate_error(&self, y1: &DVector<f64>, y2: &DVector<f64>) -> f64 {
81        let n = y1.len();
82        let mut err = 0.0f64;
83
84        for i in 0..n {
85            let tol = self.atol + self.rtol * self.state[i].abs();
86            let e = (y1[i] - y2[i]).abs() / tol;
87            err = err.max(e);
88        }
89        err
90    }
91
92    /// Computes a new step size based on the current error.
93    ///
94    /// # Arguments
95    ///
96    /// * `dt` - Current step size
97    /// * `error` - Current error estimate
98    /// * `safety` - Safety factor to prevent overly aggressive step size changes
99    ///
100    /// # Returns
101    ///
102    /// The adjusted step size for the next iteration.
103    fn compute_new_step(&self, dt: f64, error: f64, safety: f64) -> f64 {
104        let mut new_dt = dt * safety * (1.0 / error).powf(0.5);
105        new_dt = new_dt.clamp(self.min_step, self.max_step);
106        new_dt
107    }
108
109    /// Performs a single Euler-Maruyama integration step.
110    ///
111    /// # Arguments
112    ///
113    /// * `time` - Current simulation time
114    /// * `dt` - Step size
115    /// * `state` - Current state of the system (modified in-place)
116    fn euler_maruyama_step(&self, time: f64, dt: f64, state: &mut DVector<f64>) {
117        let n = state.len();
118        let mut rateiv = DVector::from_vec(vec![0.0, 0.0, 0.0]);
119        //TODO: This should be pre-calculated
120        for infusion in &self.infusions {
121            if time >= infusion.time() && time <= infusion.duration() + infusion.time() {
122                rateiv[infusion.input()] += infusion.amount() / infusion.duration();
123            }
124        }
125        let mut drift_term = DVector::zeros(n);
126        (self.drift)(
127            state,
128            &self.params,
129            time,
130            &mut drift_term,
131            rateiv,
132            &self.cov,
133        );
134
135        let mut diffusion_term = DVector::zeros(n);
136        (self.diffusion)(&self.params, &mut diffusion_term);
137
138        let mut rng = rng();
139        let normal_dist = Normal::new(0.0, 1.0).unwrap();
140
141        for i in 0..n {
142            state[i] +=
143                drift_term[i] * dt + diffusion_term[i] * normal_dist.sample(&mut rng) * dt.sqrt();
144        }
145    }
146
147    /// Solves the SDE system over the specified time interval.
148    ///
149    /// Uses adaptive step size control to balance accuracy and performance.
150    ///
151    /// # Arguments
152    ///
153    /// * `t0` - Starting time
154    /// * `tf` - Ending time
155    ///
156    /// # Returns
157    ///
158    /// A tuple containing:
159    /// * Vector of time points where solutions were computed
160    /// * Vector of state vectors corresponding to each time point
161    pub fn solve(&mut self, t0: f64, tf: f64) -> (Vec<f64>, Vec<DVector<f64>>) {
162        let mut t = t0;
163        let mut dt = self.max_step;
164        let safety = 0.9;
165        let mut times = vec![t0];
166        let mut solution = vec![self.state.clone()];
167
168        while t < tf {
169            let mut y1 = self.state.clone();
170            let mut y2 = self.state.clone();
171
172            // Single step
173            self.euler_maruyama_step(t, dt, &mut y1);
174
175            // Two half steps
176            self.euler_maruyama_step(t, dt / 2.0, &mut y2);
177            self.euler_maruyama_step(t + dt / 2.0, dt / 2.0, &mut y2);
178
179            let error = self.calculate_error(&y1, &y2);
180
181            if error <= 1.0 {
182                t += dt;
183                self.state = y2; // Use more accurate solution
184                times.push(t);
185                solution.push(self.state.clone());
186                dt = self.compute_new_step(dt, error, safety);
187                dt = dt.min(tf - t); // Don't step beyond tf
188            } else {
189                dt = self.compute_new_step(dt, error, safety);
190            }
191        }
192
193        (times, solution)
194    }
195}
