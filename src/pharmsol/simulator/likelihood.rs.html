11    std::f64::consts::FRAC_2_SQRT_PI * std::f64::consts::FRAC_1_SQRT_2 / 2.0;

25impl Predictions for SubjectPredictions {

37impl SubjectPredictions {
38    /// Calculate the likelihood of the predictions given an error model.
39    ///
40    /// This multiplies the likelihood of each prediction to get the joint likelihood.
41    ///
42    /// # Parameters
43    /// - `error_model`: The error model to use for calculating the likelihood
44    ///
45    /// # Returns
46    /// The product of all individual prediction likelihoods
47    pub fn likelihood(&self, error_model: &ErrorModel) -> f64 {
48        self.predictions
49            .iter()
50            .map(|p| p.likelihood(error_model))
51            .product()
52    }
53
54    /// Add a new prediction to the collection.
55    ///
56    /// This updates both the main predictions vector and the flat vectors.
57    ///
58    /// # Parameters

61        self.predictions.push(prediction.clone());
62        self.flat_observations.push(prediction.observation);
63        self.flat_predictions.push(prediction.prediction);
64        self.flat_time.push(prediction.time);
65    }

92/// Probability density function of the normal distribution
93fn normpdf(obs: f64, pred: f64, sigma: f64) -> f64 {
94    (FRAC_1_SQRT_2PI / sigma) * (-((obs - pred) * (obs - pred)) / (2.0 * sigma * sigma)).exp()
95}

132/// Calculate the psi matrix for maximum likelihood estimation.
133///
134/// # Parameters
135/// - `equation`: The equation to use for simulation
136/// - `subjects`: The subject data
137/// - `support_points`: The support points to evaluate
138/// - `error_model`: The error model to use
139/// - `progress`: Whether to show a progress bar
140/// - `cache`: Whether to use caching
141///
142/// # Returns
143/// A 2D array of likelihoods
144pub fn psi(
145    equation: &impl Equation,
146    subjects: &Data,
147    support_points: &Array2<f64>,
148    error_model: &ErrorModel,
149    progress: bool,
150    cache: bool,
151) -> Array2<f64> {
152    let mut pred: Array2<f64> = Array2::default((subjects.len(), support_points.nrows()).f());
153    let subjects = subjects.get_subjects();
154    let pb = match progress {
155        true => {
156            let pb = ProgressBar::new(pred.ncols() as u64 * pred.nrows() as u64);
157            pb.set_style(
158                ProgressStyle::with_template(
159                    "Simulating subjects...\n[{elapsed_precise}] {bar:40.green} {percent}% ETA:{eta}",
160                )
161                .unwrap()
162                .progress_chars("##-"),
163            );
164            Some(pb)
165        }
166        false => None,
167    };
168
169    pred.axis_iter_mut(Axis(0))
170        .into_par_iter()
171        .enumerate()
172        .for_each(|(i, mut row)| {
173            row.axis_iter_mut(Axis(0))
174                .into_par_iter()
175                .enumerate()
176                .for_each(|(j, mut element)| {
177                    let subject = subjects.get(i).unwrap();
178                    let likelihood = equation.estimate_likelihood(
179                        subject,
180                        support_points.row(j).to_vec().as_ref(),
181                        error_model,
182                        cache,
183                    );
184                    element.fill(likelihood);
185                    if let Some(pb_ref) = pb.as_ref() {
186                        pb_ref.inc(1);
187                    }
188                });
189        });
190    if let Some(pb_ref) = pb.as_ref() {
191        pb_ref.finish();
192    }
193
194    pred
195}

225/// Prediction holds an observation and its prediction
226#[derive(Debug, Clone)]
227pub struct Prediction {
228    time: f64,
229    observation: f64,
230    prediction: f64,
231    outeq: usize,
232    errorpoly: Option<(f64, f64, f64, f64)>,
233    state: Vec<f64>,
234}
235
236impl Prediction {
237    /// Get the time point of this prediction.
238    pub fn time(&self) -> f64 {
239        self.time
240    }
241
242    /// Get the observed value.
243    pub fn observation(&self) -> f64 {
244        self.observation
245    }
246
247    /// Get the predicted value.
248    pub fn prediction(&self) -> f64 {
249        self.prediction
250    }
251
252    /// Get the output equation index.
253    pub fn outeq(&self) -> usize {
254        self.outeq
255    }
256
257    /// Get the error polynomial coefficients, if available.
258    pub fn errorpoly(&self) -> Option<(f64, f64, f64, f64)> {
259        self.errorpoly
260    }
261
262    /// Calculate the raw prediction error (prediction - observation).
263    pub fn prediction_error(&self) -> f64 {
264        self.prediction - self.observation
265    }
266
267    /// Calculate the percentage error as (prediction - observation)/observation * 100.
268    pub fn percentage_error(&self) -> f64 {
269        ((self.prediction - self.observation) / self.observation) * 100.0
270    }
271
272    /// Calculate the absolute error |prediction - observation|.
273    pub fn absolute_error(&self) -> f64 {
274        (self.prediction - self.observation).abs()
275    }
276
277    /// Calculate the squared error (prediction - observation)Â².
278    pub fn squared_error(&self) -> f64 {
279        (self.prediction - self.observation).powi(2)
280    }
281
282    /// Calculate the likelihood of this prediction given an error model.
283    pub fn likelihood(&self, error_model: &ErrorModel) -> f64 {
284        let sigma = error_model.estimate_sigma(self);
285        normpdf(self.observation, self.prediction, sigma)
286    }
287
288    /// Get the state vector at this prediction point.
289    pub fn state(&self) -> &Vec<f64> {
290        &self.state
291    }
292}

297    ///
298    /// # Parameters

300    /// - `state`: The state vector at the prediction time

302    /// # Returns
